CREATE DATABASE RegistrationBot

CREATE TABLE Owners
(
    id bigint NOT NULL UNIQUE
);

CREATE TABLE Visitor 
(
    id bigint NOT NULL UNIQUE,
    name varchar NOT NULL,
    phone_number varchar NOT NULL UNIQUE
);

CREATE TABLE Business
(
   id bigint generated by default as identity primary key,
   owner_id bigint NOT NULL references Owners(id) ON DELETE CASCADE,
   name varchar NOT NULL UNIQUE,
   photo_id varchar NOT NULL UNIQUE,
   description varchar NOT NULL,
   time_of_work varchar NOT NULL
);

CREATE TABLE Employees
(
   id bigint NOT NULL UNIQUE,
   business_id bigint NOT NULL references Business(id),
   name varchar NOT NULL,
   lastname varchar NOT NULL,
   photo_id varchar NOT NULL UNIQUE,
   phone_number varchar NOT NULL UNIQUE,
   time_of_one_session varchar NOT NULL,
   description varchar NOT NULL
);

CREATE TABLE Records
(
   business_id bigint NOT NULL references Business(id),
   visitor_id bigint NOT NULL references Visitor(id),
   employee_id bigint NOT NULL references Employees(id),
   start_time timestamp NOT NULL
);
